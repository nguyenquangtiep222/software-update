-- Tạo bảng tạm thời cho bảng products
CREATE TABLE IF NOT EXISTS products_temp (
    product_id INTEGER PRIMARY KEY AUTOINCREMENT,
    product_code TEXT UNIQUE,
    product_name TEXT UNIQUE,
    product_name_ascii TEXT,
    category_id INTEGER,
    unit_id INTEGER,
    price_id INTEGER,
    stock_quantity DOUBLE, -- Chuyển đổi kiểu dữ liệu
    sold_quantity DOUBLE,  -- Chuyển đổi kiểu dữ liệu
    note TEXT,
    status INTEGER,
    status_value TEXT,
    is_deleted INTEGER,
    modified_date DATE,
    created_date DATE
);

-- Sao chép dữ liệu từ bảng products sang bảng tạm thời
INSERT INTO products_temp SELECT * FROM products;

-- Xóa bảng products cũ
DROP TABLE IF EXISTS products;

-- Đổi tên bảng tạm thời thành tên của bảng products
ALTER TABLE products_temp RENAME TO products;
